2. Add a `computeCameraPresetPositions(camera)` function that takes `camera.startPosition` and `camera.startTarget` from meta.json and computes actual 3D positions for each view:

```js
function computeCameraPresetPositions(camera) {
  if (!camera?.startPosition || !camera?.startTarget) return null;

  const [sx, sy, sz] = camera.startPosition;
  const [tx, ty, tz] = camera.startTarget;

  // Vector from target to start (the "arm")
  const dx = sx - tx;
  const dy = sy - ty;
  const dz = sz - tz;
  const armLength = Math.sqrt(dx * dx + dy * dy + dz * dz);
  const horizontalLength = Math.sqrt(dx * dx + dz * dz);

  // Azimuth angle of the start position relative to target (in XZ plane)
  const azimuth = Math.atan2(dx, dz);

  function orbitPosition(angleDegrees) {
    const angle = azimuth + (angleDegrees * Math.PI) / 180;
    return [
      tx + horizontalLength * Math.sin(angle),
      sy,
      tz + horizontalLength * Math.cos(angle),
    ];
  }

  return {
    V0: { position: [sx, sy, sz], target: [tx, ty, tz] },
    V1: { position: orbitPosition(45), target: [tx, ty, tz] },
    V2: { position: orbitPosition(90), target: [tx, ty, tz] },
    V3: { position: orbitPosition(180), target: [tx, ty, tz] },
    V4: {
      position: [tx, sy + armLength * 0.8, tz + armLength * 0.3],
      target: [tx, ty, tz],
    },
    V5: {
      position: [
        sx + (tx - sx) * 0.5,
        sy + (ty - sy) * 0.5,
        sz + (tz - sz) * 0.5,
      ],
      target: [tx, ty, tz],
    },
    V6: {
      position: [sx, ty + 0.3, sz],
      target: [tx, ty + 0.3, tz],
    },
  };
}
```

2. Implement `handleCameraPreset(presetId)`:
The lab viewer uses an iframe to display the memory scene. To communicate camera presets to the iframe, use `postMessage`:

```js
function handleCameraPreset(presetId) {
  setActivePreset(presetId);
  const positions = computeCameraPresetPositions(data?.scene);
  if (!positions?.[presetId]) return;

  const iframeEl = document.querySelector('.memory-world-lab__iframe');
  if (iframeEl?.contentWindow) {
    iframeEl.contentWindow.postMessage({
      type: 'memory-lab-camera-preset',
      preset: presetId,
      position: positions[presetId].position,
      target: positions[presetId].target,
    }, '*');
  }
}
```

Note: The iframe must have a listener to receive this. Since the memory scene route already exists with its own camera controls (OrbitControls or similar in the R3F scene), we need to add a message listener in the memory world renderer. This is task 18-02-03.
