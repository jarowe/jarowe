3. Create `src/utils/gpuCapability.js`:
```js
/**
 * Capability-based detection for gaussian splat rendering.
 * Returns true if the device can sustain splat rendering.
 * Checks: WebGL2 support, not a low-end mobile device, sufficient GPU.
 */
export function canRenderSplat() {
  // Must have WebGL2
  try {
    const canvas = document.createElement('canvas');
    const gl = canvas.getContext('webgl2');
    if (!gl) return false;

    // Check max texture size as GPU capability proxy
    const maxTexSize = gl.getParameter(gl.MAX_TEXTURE_SIZE);
    if (maxTexSize < 4096) return false;

    // Check renderer string for known low-end GPUs
    const debugInfo = gl.getExtension('WEBGL_debug_renderer_info');
    if (debugInfo) {
      const renderer = gl.getParameter(debugInfo.UNMASKED_RENDERER_WEBGL).toLowerCase();
      // Known low-end mobile GPUs
      const lowEnd = ['mali-4', 'mali-t6', 'adreno 3', 'adreno 4', 'powervr sgx', 'apple gpu']; // Apple GPU = older iPhones
      if (lowEnd.some(g => renderer.includes(g))) return false;
    }

    // Mobile heuristic: touch + small screen + low device memory
    const isMobile = 'ontouchstart' in window && window.innerWidth < 768;
    const lowMemory = navigator.deviceMemory && navigator.deviceMemory < 4;
    if (isMobile && lowMemory) return false;

    // Cleanup
    const ext = gl.getExtension('WEBGL_lose_context');
    if (ext) ext.loseContext();

    return true;
  } catch {
    return false;
  }
}
```

Key implementation details for the splat viewer:

```jsx
import * as GaussianSplats3D from '@mkkellogg/gaussian-splats-3d';
```

The viewer needs its own Three.js scene and renderer (NOT inside the existing R3F Canvas). Create a `<div ref={containerRef}>` and initialize:

```js
useEffect(() => {
  if (!capable || !containerRef.current) return;
  const viewer = new GaussianSplats3D.Viewer({
    cameraUp: [0, 1, 0],
    initialCameraPosition: [scene.cameraPosition.x, scene.cameraPosition.y, scene.cameraPosition.z],
    initialCameraLookAt: [scene.cameraTarget.x, scene.cameraTarget.y, scene.cameraTarget.z],
    rootElement: containerRef.current,
    selfDrivenMode: true, // manages its own render loop
    useBuiltInControls: true,
    dynamicScene: false,
  });
  viewer.addSplatScene(BASE + scene.splatUrl, { splatAlphaRemovalThreshold: 5 })
    .then(() => { setLoaded(true); })
    .catch(err => { console.error('[MemoryPortal] Splat load error:', err); setCapable(false); });

  return () => { viewer.dispose(); };
}, [capable, scene]);
```

For the soundtrack, create a separate Howl instance (not using AudioProvider -- the memory soundtrack is scene-specific, not the global music player):

```js
const soundRef = useRef(null);
useEffect(() => {
  if (!scene.soundtrack) return;
  const sound = new Howl({
    src: [BASE + scene.soundtrack],
    volume: 0,
    loop: true,
    onload: () => setSoundReady(true),
  });
  soundRef.current = sound;
  sound.play(); // starts muted (volume 0)
  return () => { sound.unload(); };
}, [scene.soundtrack]);

const handleUnmute = () => {
  if (!soundRef.current) return;
  soundRef.current.fade(0, 0.6, 2000); // fade in over 2s
  setMuted(false);
};
```

For narrative cards, use a ref to track which cards are visible:

const [visibleCards, setVisibleCards] = useState([]);
